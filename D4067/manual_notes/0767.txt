བསྟན་འགྱུར། སེམས་ཙམ། སི་༡༥༢ དེ་དག་དང་མཚུངས་པ་དུ་ཞེ་ན། དེ་དག་ཉིད་རང་གི་རྣམ་པར་ཤེས་པས་སྟོང་པ་སྟེ། རང་གི་རིགས་དང་མཐུན་པའི་ཕྱིར་རོ་ཞེས་བྱ་བ་ནི་རྣམ་པར་ཤེས་པ་མེད་པའི་མིག་ལ་དེ་དང་མཚུངས་པ་ཞེས་བྱའོ། ། །རྣ་བ་དང་། སྣ་དང་། ལྕེ་དང་། ལུས་ལ་ཡང་དེ་བཞིན་དུ་རིག་པར་བྱའོ། ་།ཕུང་པོ་ལྔའི་འགྲེལ་པ་སློབ་དཔོན་ཡོན་ཏན་འོད་ཀྱིས་མཛད་པ་རྫོགས་སོ། ། ཀུ་། ཡོ་རྒྱ་གར་གྱི་མཁན་པོ་ཛི་ན་མི་ཏྲ་དང་། དཱ་ན་ཤཱི་ལ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བན་དེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ་། ། :ན་བ། །#𱨊 ཀ ਮ ਨ ਨਾਲ ਹਮਲਾ
