དྷི་སཏྭ། ཧེ་ཀཱ་རུ་ཎི་ཀ་སྨ་ར་ཧྲྀ་ད་ཡཾ། ཨེ་ཧྱེ་ཧི། ཨཱརྱཱ་བ་ལོ་ཀི་ཏེ་ཤྭ་ར། པ་ར་མ་མི་ཏྲི་ཙིཏྟ་ཀཱ་རུ་ཎི་ཀ་ཀུ་རུ་ཀུ་རུ་ཀརྨ། སཱ་དྷ་ཡ་བི་དྱཱཾ། དེ་ཧི་དེ་ཧི། མེ་ཨ་རཾ། ། གཱ་མཾ་གཱ་མ། བི་ཧཾ་གཱ་མ། ། སིདྡྷ་སིདྡྷི་ཡོ་གེ་ཤྭ་ར། དུ་ཧུ་དུ་ཧུ་བཱིརྱནྟེ། མ་ཧཱ་བཱིརྱནྟེ། དྷ་ར་དྷ་ར་དྷ་ར་ཎཱི་ཤྭ་ར། ཛཱ་ལ་ཛཱ་ལ་བི་མ་ལཱ་མཱུརྟེ། ཨཱརྱ་བ་ལོ་ཤི་ཏེ་ཤྭ་ར། གྲྀཥྞ་ཛི་ན་ཛི་ཏཱ་མུ་ཀུ་ཊ་ལ་ཀྲྀ་ཏ་ཤ་རཱི་ར་ལམྦ་པྲ་ལཾ་བ་བི་ལཾ་བ། །མ་ཧཱ་སིདྡྷ། བིདྱཱི་དྷ་ར་བ་ལ་བ་ལ་མ་ཧཱ་བ་ལ་མ་ལ་མ་ལ། མ་ཧཱ་མ་ལ། ཙ་ལ་ཙ་ལ། མ་ཧཱ་ཙ་ལ་ཀྲྀཥྚ་བཀྵ། ཀྀཥྞ་བརྞྞ་གྲྀཥྞ་པཱ་ཤ་ནིརྒྷ་ཏ་ན། ཡེ་པདྨ་ཡ་སྟ། ཙ་ར་ཙ་ར་ནི་ཤྩ་རེ་ཤྭ་ར། ཀྲྀཥྞ་སརྦྦ་ཀྲྀ་ཏ་ཡ་ཛྙོ་པ་བཱི་ཏ། ཨེ་ཧྲེ་ཧི་བཱི་ར་མུ་ཁ། ཏྲི་བུ་རད་ཧ་ནེ་ཤྭ་ར། ནཱ་རཱ་ཡ་ནཱ། བ་ལ་རུ་པ་བེ་ཥ་རཱི། ཧེ་ནཱི་ལ་ཀཥྛ། ཧེ་མ་ཧཱ་ཧཱ་ལ་ཧཱ་ལ་བི་ཥ་ནི་རྫི་ཏཱ། ལོ་ཀ་སྱ་རཱ་ག་བི་ཥ་ནཱ་ཤ་ནཾ། དྭེ་ཥ་བི་ཥ་ནཱ་ཤ་ནཾ། ། མོ་ཧ་བི་ཥ་ནིར་མོཀྵ་ནཾ་ཧུ་ལུ་ཧུ་ལུ་མ་ལ། ཕ་ལ་ཕ་ལ་ཧ་རེ། མ་ཧཱ་པདྨ་ནཱ་བྷ། ས་ར་ས་ར། སི་རི་སི་རི། སུ་རུ་སུ་རུ། བུདྡྷ་བུདྡྷཾ། བོ་དྷ་བོ་དྷ། བོ་དྷ་ཡཱ་མི། མི་ཏི་ནཱི་ལ་ཀནྟུ་ཨེ་ཧེ་ཧི་ནི་ལ་ཀནྟུ། ཨེ་ཀྱེ་ཧི་བཱ་མསྦི་ཏི། ་སི་ཧ་མུ་ཁ་ཧ་ས་ཧ་ས། མུཉྫ་མུཉྩ་མ་ཧཱ་ཊྚ་ཊྚ་ཧཱ་ས་ནིརྩྭ་དེ་ནི། ཨེ་ཧེ་ཧི་བྷོ་བྷོ་མ་ཧཱ་སིདྡྷི་ཡོ་གི་ཤྭ་ར། ་བནྡྷ་བནྡྷ། བ་ཙི། ། སཱ་དྷ་ཡ་སཱ་དྷ་ཡ། བིདྡྷཾ། སྨ་ར་སྨ་ར་ཏྭཾ། བྷ་ག་བནྟི། ལོ་ཀ་བི་ལོ་ཀི་ཏཱཾ། ཏ་ཐཱ་ག་ཏཱ། ད་དཱ་ཧི་མེ་དརྟ་ནཾ། ཀ་མ་སྱ་དནྟ་ནཾ། པྲ་ཝཱ་ད་ཡ་མ་ན་སཱ་ཧཱ། སིདྡྷཱ་ཡ་སྭཱ་ཧཱ། མ་ཧཱ་སིདྡྷཱ་ཡ་སྭཱ་ཧཱ། སིདྷ་ཡོ་གེ་ཕྱོགས་རཱ་ཡ་སྭཱ་ཧཱ། ནཱི་ལ་ཀཏྟཱ་ཡ་སྭཱ་ཧཱ། བ་རུ་ཧ་མུ་ཁཱ་ཡ་སྭཱ་ཧཱ། ། སིདྷ་མུ་ཁཱ་ཡ་སྭཱ་ཧཱ། མ་ཧཱ་མ་ནཱ་རི་སི་ཧ་མུ་ཁཱ་ཡ་སྭཱ་ཧཱ། །: སིདྡྷ་བིཅུ་དྷ་རཱ་ཡ་སྭཱ་ཧཱ། ། པདྨ་ཧ་སྣ་ཡ་སྭཱ་ཧཱ། མ་ཧཱ་པདྨ་ཧ་ལྟ་ཡ་སྭཱ་ཧཱ། བཛྲ་ཧ་ལྟ་ཡ་སྭཱ་ཧཱ། ཀྲྀཥྚ་སརྦ་ཀྲྀ་ཏ་ཡ་དྷོ་པ་བཱི་ཏ་ཡ་སྭཱ་ཧཱ། ། མ་ཧཱ་ཀཱ་བ་མུ་ཀུ་ཊ་དྷ་རཱ་ཡ་སྭཱ་ཧཱ། ཙཀྲ་ཨ་ཡུ་དྷ་དྷ་རཱ་ཡ་སྭཱ་ཧཱ། ཤཾ་ཁ་ཤརྡ་ནིནཱ་ད་ན་ཀ་རཱ་ཡ་སྭཱ་ཧཱ། ནི་བོ་དྷ་ན་ཀ་རཱ་ཡ་སྭཱ་ཧཱ། བཱ་མ་སྐནྡྷ་དེ་ཤམྦི་ཏ་གིཥཱ་ཛ་ནཱ་ཡ་སྭཱ་ཧཱ། བཱ་མ་ཧ་སྟ་བྱ་གྲྭ་ཙརྨ་ནི་བས་ནཱ་ཡ་སྭཱ་ཧཱ། ལོ་ཀེ་ཤྭ་རཱ་ཡ་སྭཱ་ཧཱ། མ་ཧཱ་ལོ་ཀ་ཤྭ་རཱ་ཡ་སྭཱ་ཧཱ། ། སརྦ་སིདྡྷེ་ཤྭ་རཱ་ཡ་སྭཱ་ཧཱ ན་མོ་བྷ་ག་བ་ཏེ། ཨཱརྱ་བ་ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ། བྷོ་དྷི་སཏྭཱ་ཡ་མ་ཧཱ་སཏྭཱ་ཡ།
